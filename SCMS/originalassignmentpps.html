1. Program to Determine Profit or Loss

#include <stdio.h> int main() { float costPrice, sellingPrice; printf("Enter Cost Price: "); scanf("%f", &costPrice); printf("Enter Selling Price: "); scanf("%f", &sellingPrice); if (sellingPrice > costPrice) { printf("Profit = %.2f\n", sellingPrice - costPrice); } else if (costPrice > sellingPrice) { printf("Loss = %.2f\n", costPrice - sellingPrice); } else { printf("No Profit, No Loss.\n"); } return 0; }

2. Swap Two Numbers Without Third Variable

#include <stdio.h> int main() { int a, b; printf("Enter first number: "); scanf("%d", &a); printf("Enter second number: "); scanf("%d", &b); a = a + b; b = a - b; a = a - b; printf("After swapping:\n"); printf("First number = %d\n", a); printf("Second number = %d\n", b); return 0; }
Logic Used:
  • a = a + b
  • b = a - b
  • a = a - b

3. Check Divisibility by 5 (Ternary Operator)

#include <stdio.h> int main() { int num; printf("Enter a number: "); scanf("%d", &num); (num % 5 == 0) ? printf("The number is divisible by 5.\n") : printf("The number is not divisible by 5.\n"); return 0; }
